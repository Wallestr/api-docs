# --- Файл: utils/maxgram_api.py ---
import aiohttp
import logging

API_KEY = "ВАШ_API_КЛЮЧ_БОТА"
URL = "https://traff.oudap-spt.ru/get-sponsors"

async def get_maxgram_sponsors(user_id: int):
    """Универсальная функция для запроса спонсоров."""
    headers = { "Auth": API_KEY }
    payload = { "user_id": user_id}

    
    async with aiohttp.ClientSession() as session:
        try:
            async with session.post(URL, headers=headers, json=payload, timeout=10) as response:
                return await response.json()
        except Exception as e:
            logging.error(f"Ошибка запроса к maxgram API: {e}")
            return None
